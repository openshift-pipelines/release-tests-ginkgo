**Create `tests/triggers/cron_test.go`** with 1 test:

PIPELINES-04-TC01 has a unique flow:
1. Create 4 YAML resources (pipeline, triggerbinding, triggertemplate, eventlistener)
2. Expose EventListener "cron-listener"
3. Verify image stream "golang" exists (use `oc` CLI check)
4. Create cron job with schedule "*/1 * * * *" (use pkg/k8s.CreateCronJob)
5. Watch for pipelinerun resources (use pkg/pipelines.WatchForPipelineRun)
6. Delete cron job (use pkg/k8s.DeleteCronJob)
7. Assert no new pipelineruns created (use pkg/pipelines.AssertForNoNewPipelineRunCreation)
8. Cleanup Triggers

```go
package triggers_test

import (
    . "github.com/onsi/ginkgo/v2"
    "github.com/openshift-pipelines/release-tests-ginkgo/pkg/k8s"
    "github.com/openshift-pipelines/release-tests-ginkgo/pkg/oc"
    "github.com/openshift-pipelines/release-tests-ginkgo/pkg/pipelines"
    "github.com/openshift-pipelines/release-tests-ginkgo/pkg/triggers"
)

var _ = Describe("CronJob Triggers", Label("triggers"), func() {
    It("PIPELINES-04-TC01: Create Triggers using k8s cronJob", Label("e2e", "triggers", "non-admin", "sanity"), func() {
        ns := sharedClients.Namespace
        oc.Create("testdata/triggers/cron/example-pipeline.yaml", ns)
        oc.Create("testdata/triggers/cron/triggerbinding.yaml", ns)
        oc.Create("testdata/triggers/cron/triggertemplate.yaml", ns)
        oc.Create("testdata/triggers/cron/eventlistener.yaml", ns)
        DeferCleanup(func() { triggers.CleanupTriggers(sharedClients, "cron-listener", ns) })
        triggers.ExposeEventListener(sharedClients, "cron-listener", ns)
        // Verify image stream exists (oc get is -n openshift golang)
        // Use cmd.MustSucceed or a helper
        k8s.CreateCronJob(sharedClients, "*/1 * * * *", ns)
        DeferCleanup(func() { k8s.DeleteCronJob(sharedClients, ns) })
        pipelines.WatchForPipelineRun(sharedClients, ns)
        k8s.DeleteCronJob(sharedClients, ns)
        pipelines.AssertForNoNewPipelineRunCreation(sharedClients, ns)
    })
})
```
