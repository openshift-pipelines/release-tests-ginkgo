Describe("PIPELINES-30-TC01: Configure PAC with push and pull_request events", Ordered, Label("pac", "sanity", "e2e"), func() {
    var (
        gitlabClient *gitlab.Client
        project      *gitlab.Project
        projectID    int
        mrID         int
        smeeURL      string
        namespace    string
    )

    BeforeAll(func() {
        namespace = store.Namespace()
        gitlabClient = pac.InitGitLabClient()
        pac.SetGitLabClient(gitlabClient)

        var err error
        smeeURL, err = pac.SetupSmeeDeployment(sharedClients, namespace)
        Expect(err).NotTo(HaveOccurred())

        project, err = pac.SetupGitLabProject(sharedClients, namespace, ...)

        projectID = project.ID

        DeferCleanup(func() {
            err := pac.CleanupPAC(sharedClients, namespace, projectID, "gosmee-client")
            Expect(err).NotTo(HaveOccurred())
        })
    })

    It("should generate pull_request PipelineRun YAML", func() { ... })
    It("should generate push PipelineRun YAML", func() { ... })
    It("should configure preview changes with both files", func() { ... })
    It("should validate pull_request PipelineRun succeeds", func() { ... })
    It("should trigger push event on main branch", func() { ... })
    It("should validate push PipelineRun succeeds", func() { ... })
})

**Key patterns for all three tests:**

- Declare state variables (projectID, mrID, etc.) at the Describe level -- this is safe in Ordered containers

- Use `Expect(err).NotTo(HaveOccurred())` for error assertions
- Use `Eventually` for polling PipelineRun status instead of `time.Sleep`
- Use `Consistently` for asserting 0 pipelineruns (TC02 step 4)
- Each It block name should describe the action clearly
- `sharedClients` comes from suite_test.go package variable
